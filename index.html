Connect your Google Sheet

Takes about 5 minutes. After this, every entry you log syncs to one Sheet — accessible from any device.

① Create a Google Sheet

Open sheets.new ↗ and leave it blank. Note the Sheet ID from the URL:
docs.google.com/spreadsheets/d/THIS_PART/edit

② Add the Apps Script

In your Sheet, go to Extensions → Apps Script. Delete any existing code, paste the script below, then click Save.

function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); var data = JSON.parse(e.postData.contents); var HEADERS = ['date','entry_mode','sleep_time','wake_time','sleep_hours', 'mood','habits','social','work_hours','steps', 'food_breakfast','food_lunch','food_dinner','food_snacks', 'content_types','screen_hours','activities','note','saved_at']; if (data.action === 'init') { if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); return out({ok:true}); } if (data.action === 'append') { sheet.appendRow(rowArr(data.row, HEADERS)); return out({ok:true}); } if (data.action === 'update') { var vals = sheet.getDataRange().getValues(); for (var i=1;i<vals.length;i++) { if (String(vals[i][0])===String(data.date)) { sheet.getRange(i+1,1,1,HEADERS.length) .setValues([rowArr(data.row,HEADERS)]); return out({ok:true,updated:true}); } } sheet.appendRow(rowArr(data.row,HEADERS)); return out({ok:true,updated:false}); } if (data.action === 'delete') { var vals = sheet.getDataRange().getValues(); for (var i=1;i<vals.length;i++) { if (String(vals[i][0])===String(data.date)) { sheet.deleteRow(i+1); return out({ok:true}); } } return out({ok:false}); } } function doGet() { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getActiveSheet(); var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return out({entries:[]}); var h = rows[0]; var entries = rows.slice(1).map(function(r){ var o={}; h.forEach(function(k,i){o[k]=r[i];}); return o; }); return out({entries:entries}); } function rowArr(r,h){return h.map(function(k){return r[k]||''});} function out(obj){ return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
③ Deploy as Web App

Click Deploy → New deployment. Choose type Web app. Set:
• Execute as: Me
• Who has access: Anyone

Click Deploy, authorise when prompted, and copy the deployment URL.
